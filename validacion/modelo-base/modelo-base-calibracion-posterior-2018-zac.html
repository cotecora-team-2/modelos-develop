Verificación posterior para total de votos

set.seed(83441)
candidato <- "JAMK"
if(FALSE){
  ajuste_muestras_2 <- map(1:150, function(rep){
  muestra_1 <- obtener_muestra_marco(datos_ent, frac = 0.12)
  sim_datos$y_f <- datos_ent[[candidato]]
  muestra_1$y <- muestra_1[["votos"]][[candidato]] 
  datos_1 <- c(parametros, muestra_1, sim_datos)
  total <- datos_1$y_f %>% sum()
  ajuste <- modelo$sample(data = datos_1, 
                          seed = 2131,
                          iter_sampling = 1000, iter_warmup = 1000,
                          refresh = 0, 
                          chains = 4, 
                          adapt_delta = 0.97,
                          max_treedepth = 12,
                          parallel_chains = 4,
                          show_messages = FALSE)
  y_out_df <- ajuste$draws() %>% as_draws_df()
  y_out_df %>% select(y_out) %>% 
    mutate(rep = rep, total = total)
  })
  write_rds(ajuste_muestras_2, "salidas/ajuste-posterior-muestras-2-zac.rds")
} else {
  ajuste_muestras_2 <- read_rds("salidas/ajuste-posterior-muestras-2-zac.rds")
}
ajuste_resumen_tbl <- ajuste_muestras_2 %>% bind_rows() %>% 
  group_by(rep) %>% 
  summarise(sup = quantile(y_out, 0.975), inf = quantile(y_out, 0.025),
            total = first(total))
ggplot(ajuste_resumen_tbl, aes(x = rep, ymin = inf, ymax = sup)) +
  geom_hline(aes(yintercept = total), colour = "salmon") +
  geom_linerange() 

ajuste_resumen_tbl %>% 
  summarise(cobertura = mean(total >= inf & total <= sup), n = n(), error_porc = 100 - 100*mean((sup+inf)/(2*total))) %>% 
  mutate(error = 2*sqrt(cobertura*(1-cobertura)/n))
